A loop with symmetric bounds which calculates a number's factorial 

cout << "Enter a number: ";
int n;
cin >> n;
double factorial = 1;
for (i = 1; i <= n; i++)
   factorial = factorial * i

An asymmetric bounded loop which determines the number of a's in a word 

string s;
cout << "Enter a word";
cin >> s;
int len = s.length();
int count_a = 0;
int i;
for (i = 0; i < len; i++)
  if (s.substr(i,1) == "a") 
    count_a++;
