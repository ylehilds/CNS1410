int a;            // global scope
int b;            // global scope

int f(int c)      // local scope to f
{  
   int n = 0;     // local scope to f
   a = c;
   if (n < c)
      n = a + b;
   return n;
}

int g(int c)     // local scope to f
{  
   int n = 0;    // local scope to f
   int a = c;    // local scope to f
   if (n < f(c))
      n = a + b;
   return n;
}

int main()
{  
   int i = 1;      // local scope to main
   int b = g(i);   // local scope to main
   cout << a + b + i << "\n";
   return 0;
}

The output of this program is 3.
